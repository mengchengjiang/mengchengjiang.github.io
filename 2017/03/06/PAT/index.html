    if alpha == 'P':

        sumP += 1
    elif alpha == 'A':
        mark[index] = (0,1,0)
        sumA += 1
    else:
        mark[index] = (0,0,1)
        sumT += 1

    index += 1

index = 0
sumMark = [0]*len(string)
for alpha in string:

    if alpha == 'P':
        if index == 0:
            sumMark[index] = (1,0,sumT)
        else:
            sumMark[index] = (sumMark[index-1][0] + 1 ,0,sumMark[index-1][2])
    elif alpha == 'A':
        if index == 0:
            sumMark[index] = (0,0,sumT)
        else:
            sumMark[index] = (sumMark[index-1][0],1,sumMark[index -1][2])
    else:
        if index == 0:
            sumMark[index] = (0,0,sumT - 1)
        else:

    index += 1
